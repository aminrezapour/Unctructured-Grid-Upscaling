#GPRS for Linux
Note that Intel mkl libraries are required!

#dirs:
bin - GPRS executable: gprs08Linux
CODE - source code and makefile
Samples - GPRS samples, see the "readme.txt" in this directory for the detailed
   descriptions of samples

#build GPRS
In the CODE directory, set the micros GPRS & MKLROOT in Makefile. Once you set these micros, 
then type make to build the GPRS.

#to run gprs08Linux, set path of Intel MKL 64 bit libs and SAMG lib into LD_LIBRARY_PATH,
for example

#set mkl
setenv LD_LIBRARY_PATH /usr/local/INTEL/Compiler/11.1/069/mkl/lib/em64t:"$LD_LIBRARY_PATH"

#set samg
setenv SAMG_HOME $HOME/GPRS08Linux/LIBS/SAMG
setenv LD_LIBRARY_PATH "$SAMG_HOME":"$SAMG_HOME"/ifort64.9.1.041-emt64/lib:"$LD_LIBRARY_PATH"
